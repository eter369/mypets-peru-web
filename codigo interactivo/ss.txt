<script>
  const answers = {};
  let currentStep = 1;
  const totalSteps = 4;

  // Selección de opciones
  document.querySelectorAll('.option').forEach(opt => {
    opt.addEventListener('click', () => {
      const key = opt.dataset.key;
      // limpia hermanos
      opt.parentElement.querySelectorAll('.option').forEach(o => o.classList.remove('selected'));
      opt.classList.add('selected');
      answers[key] = opt.dataset.value;
      document.getElementById('btn-next').disabled = false;
    });
  });

  // Navegación
  document.getElementById('btn-next').addEventListener('click', () => {
    if (currentStep < totalSteps) {
      goToStep(currentStep + 1);
    } else {
      showResult();
    }
  });
  document.getElementById('btn-back').addEventListener('click', () => {
    if (currentStep > 1) goToStep(currentStep - 1);
  });

  function goToStep(step) {
    document.querySelectorAll('.quiz-step').forEach(s => s.classList.remove('active'));
    document.querySelector(`.quiz-step[data-step="${step}"]`).classList.add('active');
    currentStep = step;

    // progress dots
    const dots = document.querySelectorAll('.progress-dot');
    dots.forEach((d, i) => {
      d.classList.remove('active','done');
      if (i + 1 < step) d.classList.add('done');
      else if (i + 1 === step) d.classList.add('active');
    });

    // botones
    document.getElementById('btn-back').disabled = step === 1;
    const stepKey = ['space','time','kids','vibe'][step-1];
    document.getElementById('btn-next').disabled = !answers[stepKey];
    document.getElementById('btn-next').innerHTML = step === totalSteps ? 'Ver resultado ✨' : 'Siguiente →';
  }

  // Lógica de matching simple
  function calculateMatch() {
    const {space, time, kids, vibe} = answers;
    // Bulldog Francés: ideal apt, calm, low/mid time, kids ok
    // Bulldog Inglés: tranquilo, casa pequeña, kids ok, leal
    // Pomerania: pequeño, apt, activo, no apto niños muy pequeños
    // Salchicha: leal, casa, jardín, juguetón

    let scores = {
      'Bulldog Francés Exótico':0,
      'Bulldog Inglés Exótico':0,
      'Pomerania Exótico':0,
      'Salchicha Exótico':0
    };

    if (space==='apt'){ scores['Bulldog Francés Exótico']+=3; scores['Pomerania Exótico']+=3; }
    if (space==='house-small'){ scores['Bulldog Inglés Exótico']+=3; scores['Bulldog Francés Exótico']+=2; }
    if (space==='house-big'){ scores['Salchicha Exótico']+=3; scores['Bulldog Inglés Exótico']+=2; }

    if (time==='low'){ scores['Bulldog Inglés Exótico']+=3; scores['Bulldog Francés Exótico']+=2; }
    if (time==='mid'){ scores['Bulldog Francés Exótico']+=2; scores['Pomerania Exótico']+=2; }
    if (time==='high'){ scores['Salchicha Exótico']+=3; scores['Pomerania Exótico']+=2; }

    if (kids==='yes'){ scores['Bulldog Francés Exótico']+=3; scores['Bulldog Inglés Exótico']+=3; scores['Pomerania Exótico']-=2; }
    if (kids==='teens'){ scores['Salchicha Exótico']+=2; scores['Pomerania Exótico']+=2; scores['Bulldog Francés Exótico']+=1; }
    if (kids==='no'){ scores['Pomerania Exótico']+=2; scores['Salchicha Exótico']+=1; }

    if (vibe==='calm'){ scores['Bulldog Inglés Exótico']+=3; scores['Bulldog Francés Exótico']+=2; }
    if (vibe==='active'){ scores['Pomerania Exótico']+=3; scores['Salchicha Exótico']+=2; }
    if (vibe==='loyal'){ scores['Salchicha Exótico']+=3; scores['Bulldog Francés Exótico']+=1; }

    return Object.entries(scores).sort((a,b)=>b[1]-a[1])[0][0];
  }

  const breedInfo = {
    'Bulldog Francés Exótico':{
      emoji:'🐾',
      desc:'Compañero ideal: cariñoso, perfecto para departamentos y excelente con niños. Bajo nivel de actividad, muy sociable y fácil de cuidar.'
    },
    'Bulldog Inglés Exótico':{
      emoji:'🦴',
      desc:'Tranquilo, leal y muy paciente con niños. Necesita poco ejercicio, perfecto para hogares relajados que buscan un compañero calmado.'
    },
    'Pomerania Exótico':{
      emoji:'✨',
      desc:'Pequeño, vivaz y juguetón. Ideal para departamentos y dueños activos que disfrutan jugar e interactuar a diario con su mascota.'
    },
    'Salchicha Exótico':{
      emoji:'🌟',
      desc:'Leal, divertido y muy apegado a su familia. Le encanta jugar y explorar; perfecto para casas con espacio y dueños activos.'
    }
  };

  function showResult() {
    const breed = calculateMatch();
    document.getElementById('result-breed').innerText = breed;
    document.getElementById('result-emoji').innerText = breedInfo[breed].emoji;
    document.getElementById('result-desc').innerText = breedInfo[breed].desc;

    document.querySelectorAll('.quiz-step').forEach(s => s.classList.remove('active'));
    document.getElementById('quiz-nav').style.display = 'none';
    document.getElementById('result').classList.add('show');

    // todos los dots como done
    document.querySelectorAll('.progress-dot').forEach(d => {
      d.classList.remove('active');
      d.classList.add('done');
    });
  }

  function restartQuiz(){
    Object.keys(answers).forEach(k => delete answers[k]);
    document.querySelectorAll('.option.selected').forEach(o => o.classList.remove('selected'));
    document.getElementById('result').classList.remove('show');
    document.getElementById('quiz-nav').style.display = 'flex';
    goToStep(1);
  }
</script>
